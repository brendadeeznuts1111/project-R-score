// Import the library
import { shortcuts } from 'keyboard-shortcuts-lite';

// Register shortcuts
shortcuts.register('ctrl+s', () => {
    console.log('Save triggered!');
    saveFile();
});

shortcuts.register('ctrl+n', () => {
    console.log('New file created');
    createNewFile();
});

shortcuts.register('ctrl+b', () => {
    document.execCommand('bold');
});

// Initialize the system
shortcuts.init();

// React Hook Integration
import { useKeyboardShortcuts } from './useKeyboardShortcuts';

const TextEditor = () => {
    const [content, setContent] = useState('');
    const { registerShortcut } = useKeyboardShortcuts();
    
    useEffect(() => {
        registerShortcut('ctrl+s', () => {
            saveContent();
        });
        
        registerShortcut('ctrl+z', () => {
            undo();
        });
    }, [registerShortcut]);
    
    return (
        <textarea
            value={content}
            onChange={(e) => setContent(e.target.value)}
        />
    );
};

// Vue Composable
import { useKeyboardShortcuts } from './useKeyboardShortcuts';

export default {
    setup() {
        const content = ref('');
        const { register } = useKeyboardShortcuts();
        
        onMounted(() => {
            register('ctrl+s', () => {
                saveContent();
            });
            
            register('ctrl+r', () => {
                refreshData();
            });
        });
        
        return {
            content
        };
    }
};

// Composable implementation
export function useKeyboardShortcuts() {
    const manager = shortcuts;
    
    onMounted(() => manager.init());
    onUnmounted(() => manager.destroy());
    
    return {
        register: manager.register.bind(manager),
        unregister: manager.unregister.bind(manager)
    };
}

// Performance Monitoring
class PerformanceMonitor {
    constructor() {
        this.metrics = {
            registrations: 0,
            executions: 0,
            averageTime: 0,
            errors: 0
        };
        
        this.setupTracking();
    }
    
    setupTracking() {
        // Track registration time
        const originalRegister = shortcuts.register;
        shortcuts.register = (key, callback, options) => {
            const start = performance.now();
            const result = originalRegister(key, callback, options);
            const end = performance.now();
            
            this.metrics.registrations++;
            console.log(`Registered ${key} in ${(end-start).toFixed(2)}ms`);
            return result;
        };
    }
    
    getMetrics() {
        return this.metrics;
    }
}

const monitor = new PerformanceMonitor();
