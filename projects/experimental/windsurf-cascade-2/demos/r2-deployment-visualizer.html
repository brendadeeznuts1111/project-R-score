Complete Deployment Script
One-click deployment script that handles everything from package creation to global distribution 
#!/usr/bin/env bun
// Complete R2 Deployment Script

import { execSync } from 'child_process';
import { readFileSync, writeFileSync } from 'fs';

console.log('üöÄ Starting Global Trading System Deployment...');

// Step 1: Configure packages
console.log('üì¶ Configuring packages...');
const packageJson = {
  name: 'global-trading-system',
  version: '1.0.0',
  type: 'module',
  dependencies: {
    '@cloudflare/r2': '^3.0.0',
    'hono': '^4.0.0',
    'bun-types': '^1.0.0'
  }
};

writeFileSync('package.json', JSON.stringify(packageJson, null, 2));

// Step 2: Install dependencies
console.log('‚¨áÔ∏è Installing dependencies...');
execSync('bun install', { stdio: 'inherit' });

// Step 3: Build packages
console.log('üî® Building packages...');
execSync('bun run build:all', { stdio: 'inherit' });

// Step 4: Deploy to R2
console.log('‚òÅÔ∏è Deploying to Cloudflare R2...');
execSync('bun run deploy:r2', { stdio: 'inherit' });

// Step 5: Verify deployment
console.log('‚úÖ Verifying deployment...');
execSync('bun run verify:deployment', { stdio: 'inherit' });

console.log('üéâ Global Trading System deployed successfully!');
console.log('üåç Available at: https://global-trading.your-domain.com');
