// React Hook Integration
import { useKeyboardShortcuts } from './useKeyboardShortcuts';

const TextEditor = () => {
    const [content, setContent] = useState('');
    const { registerShortcut } = useKeyboardShortcuts();
    
    useEffect(() => {
        // Register shortcuts for this component
        const unregisterSave = registerShortcut('ctrl+s', () => {
            saveContent(content);
        }, {
            description: 'Save content',
            category: 'file'
        });
        
        const unregisterUndo = registerShortcut('ctrl+z', () => {
            undo();
        }, {
            description: 'Undo action',
            category: 'edit'
        });
        
        // Automatic cleanup on unmount
        return () => {
            unregisterSave();
            unregisterUndo();
        };
    }, [registerShortcut, content]);
    
    return (
        <div className="text-editor">
            <textarea
                value={content}
                onChange={(e) => setContent(e.target.value)}
                placeholder="Type here... (Ctrl+S to save)"
            />
        </div>
    );
};

// Context Provider
export const KeyboardShortcutsProvider = ({ children }) => {
    const [manager] = useState(() => shortcuts);
    
    useEffect(() => {
        manager.init();
        return () => manager.destroy();
    }, [manager]);
    
    return (
        <KeyboardContext.Provider value={{ manager }}>
            {children}
        </KeyboardContext.Provider>
    );
};

// Vue Composable
import { useKeyboardShortcuts } from './useKeyboardShortcuts';

export default {
    setup() {
        const content = ref('');
        const { register } = useKeyboardShortcuts();
        
        onMounted(() => {
            // Register shortcuts
            register('ctrl+s', () => {
                saveContent(content.value);
            }, {
                description: 'Save content',
                category: 'file'
            });
            
            register('ctrl+r', () => {
                refreshData();
            }, {
                description: 'Refresh data',
                category: 'data'
            });
            
            register('ctrl+e', () => {
                toggleEditMode();
            }, {
                description: 'Toggle edit mode',
                category: 'view'
            });
        });
        
        // Automatic cleanup
        onUnmounted(() => {
            // Cleanup handled by composable
        });
        
        return {
            content,
            saveContent: () => console.log('Saving...'),
            refreshData: () => console.log('Refreshing...'),
            toggleEditMode: () => console.log('Toggle edit mode')
        };
    }
};

// Composable implementation
export function useKeyboardShortcuts() {
    const manager = shortcuts;
    const shortcuts = ref([]);
    
    onMounted(() => {
        manager.init();
    });
    
    onUnmounted(() => {
        manager.destroy();
    });
    
    const register = (key, callback, options = {}) => {
        const unregister = manager.register(key, callback, options);
        shortcuts.value.push({ key, options });
        return unregister;
    };
    
    return {
        register,
        shortcuts: readonly(shortcuts),
        manager
    };
}

// Performance Monitoring Class
class PerformanceMonitor {
    constructor() {
        this.metrics = {
            registrations: 0,
            executions: 0,
            averageTime: 0,
            errors: 0,
            memoryUsage: 0
        };
        
        this.setupTracking();
    }
    
    setupTracking() {
        // Track registration performance
        const originalRegister = shortcuts.register;
        shortcuts.register = (key, callback, options) => {
            const start = performance.now();
            const result = originalRegister.call(shortcuts, key, callback, options);
            const end = performance.now();
            
            this.metrics.registrations++;
            console.log(`Registered ${key} in ${(end-start).toFixed(2)}ms`);
            return result;
        };
        
        // Track execution performance
        shortcuts.on('execute', (data) => {
            const start = performance.now();
            this.metrics.executions++;
            
            setTimeout(() => {
                const end = performance.now();
                const executionTime = end - start;
                this.metrics.averageTime = 
                    (this.metrics.averageTime + executionTime) / 2;
                
                this.updateChart(executionTime);
            }, 0);
        });
    }
    
    runStressTest(count = 1000) {
        const start = performance.now();
        
        // Register many shortcuts rapidly
        for (let i = 0; i < count; i++) {
            shortcuts.register(`ctrl+shift+${i}`, () => {
                console.log(`Stress test ${i}`);
            });
        }
        
        const end = performance.now();
        console.log(`Stress test: ${count} shortcuts in ${(end-start).toFixed(2)}ms`);
        
        // Memory check
        if (performance.memory) {
            this.metrics.memoryUsage = 
                performance.memory.usedJSHeapSize / 1024 / 1024;
        }
    }
    
    getMetrics() {
        return { ...this.metrics };
    }
}

const monitor = new PerformanceMonitor();

// Enterprise Manager
class EnterpriseShortcutManager extends KeyboardShortcutManager {
    constructor(config) {
        super(config);
        this.initializeEnterpriseFeatures();
    }
    
    initializeEnterpriseFeatures() {
        // User management
        this.userManager = new UserManager();
        
        // Security
        this.securityManager = new SecurityManager();
        
        // Analytics
        this.analyticsManager = new AnalyticsManager();
        
        // Persistence
        this.persistenceManager = new PersistenceManager();
    }
    
    registerEnterpriseShortcut(key, callback, options = {}) {
        // Security check
        if (!this.securityManager.hasPermission(options.permission)) {
            throw new Error('Insufficient permissions');
        }
        
        // Audit logging
        this.analyticsManager.logRegistration(key, options);
        
        // Register with enhanced features
        return this.register(key, callback, {
            ...options,
            enterprise: true,
            audit: true
        });
    }
    
    generateReport() {
        return {
            usage: this.analyticsManager.getUsageReport(),
            security: this.securityManager.getSecurityReport(),
            performance: this.getPerformanceReport(),
            users: this.userManager.getUserReport()
        };
    }
}
