Button Component

Use data-action attribute for JS hooks, not class names

<!-- Use data-action for JS hooks, not class names -->
<button data-action="clear-alerts" class="btn btn-secondary btn-sm" aria-label="Clear all alerts">
  Clear
</button>

<button data-action="close-fraud-alert" class="btn btn-danger btn-full" aria-label="Acknowledge alert">
  Acknowledge Alert
</button>
