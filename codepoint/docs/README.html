// Dependency injection container to prevent circular dependencies
class DIContainer {
  private services = new Map<string, any>();
  private factories = new Map<string, () => any>();

  register<T>(name: string, factory: () => T): void {
    this.factories.set(name, factory);
  }

  get<T>(name: string): T {
    if (!this.services.has(name)) {
      const factory = this.factories.get(name);
      if (!factory) {
        throw new Error(`Service not registered: ${name}`);
      }
      this.services.set(name, factory());
    }
    return this.services.get(name);
  }

  // Initialize all core modules
  initialize(): void {
    // Register core services (0x0000-0x00FF)
    this.register('config', () => new ConfigService());
    this.register('logger', () => new LoggerService());
    this.register('metrics', () => new MetricsService());

    // Register protocol services (0x0100-0x01FF)
    this.register('httpProxy', () => new HTTPProxy(this.get('logger')));
    this.register('wsProxy', () => new WebSocketProxy(
      this.get('logger'),
      this.get('metrics')
    ));

    // Register pattern services (0x0200-0x02FF)
    this.register('circuitBreaker', () => new CircuitBreaker(
      this.get('config'),
      this.get('logger')
    ));

    // Register security services (0x0300-0x03FF)
    this.register('authenticator', () => new Authenticator(
      this.get('config')
    ));
  }
}

// Usage
const container = new DIContainer();
container.initialize();

const httpProxy = container.get<HTTPProxy>('httpProxy');
const circuitBreaker = container.get<CircuitBreaker>('circuitBreaker');

// Module registry for dynamic loading
class ModuleRegistry {
  private static registry = new Map<number, {
    path: string;
    exports: Record<string, any>;
  }>();

  static register(codePoint: number, path: string, exports: Record<string, any>) {
    this.registry.set(codePoint, { path, exports });
  }

  static get(codePoint: number): any {
    const entry = this.registry.get(codePoint);
    if (!entry) {
      throw new Error(`Module not found for code point: 0x${codePoint.toString(16)}`);
    }
    return entry.exports;
  }

  static getAll(): Map<number, string> {
    const result = new Map<number, string>();
    for (const [codePoint, entry] of this.registry.entries()) {
      result.set(codePoint, entry.path);
    }
    return result;
  }
}

// Register modules
ModuleRegistry.register(0x01, "@bun/proxy", { createProxyServer });
ModuleRegistry.register(0x100, "@bun/proxy/protocols/http", { HTTPProxy });
ModuleRegistry.register(0x200, "@bun/proxy/patterns/circuit-breaker", { CircuitBreaker });

// Usage
const proxyFactory = ModuleRegistry.get(0x01);
const server = proxyFactory.createProxyServer(config);

class TableBuilder {
  constructor() {
    this.columns = [];
    this.rows = [];
    this.options = {};
  }

  addColumn(key, header, type = "string", options = {}) {
    this.columns.push({ key, header, type, ...options });
    return this;
  }

  addRow(data) {
    this.rows.push(data);
    return this;
  }

  setOption(key, value) {
    this.options[key] = value;
    return this;
  }

  build() {
    return inspect.table(
      { columns: this.columns, rows: this.rows },
      this.options
    );
  }

  render() {
    console.log(this.build());
  }
}

// Usage
new TableBuilder()
  .addColumn("name", "Name", "string", { width: 20 })
  .addColumn("status", "Status", "badge")
  .addColumn("latency", "Latency", "number", { format: v => `${v}ms` })
  .addRow({ name: "HTTP Proxy", status: "connected", latency: 28 })
  .addRow({ name: "WebSocket", status: "connecting", latency: 125 })
  .setOption("theme", "dark")
  .setOption("zebra", true)
  .render();
